cmake_minimum_required(VERSION 3.13)

include($ENV{PICO_SDK_PATH}/external/pico_sdk_import.cmake)

project(control C CXX ASM)
set(CMAKE_C_STANDARD 11)

pico_sdk_init()

add_executable(control
    control.c
)

target_link_libraries(control
    pico_stdlib
    hardware_adc
)

# Habilitar salida por USB, deshabilitar UART
pico_enable_stdio_usb(control 1)
pico_enable_stdio_uart(control 0)

# No bloquear esperando conexión USB
target_compile_definitions(control PRIVATE
    PICO_STDIO_USB_CONNECT_WAIT_TIMEOUT_MS=0
)

pico_add_extra_outputs(control)
